% Effective Branching Factor (EBF) equation
\begin{equation}
    N(d) = \sum_{i=0}^{d} (b^*)^i = \frac{(b^*)^{d+1} - 1}{b^* - 1}
\end{equation}
\begin{equation}
    b^* \approx \frac{N_d}{N_{d-1}}
\end{equation}
% Where:
% N(d) is the total nodes evaluated at depth d.
% b^* is the effective branching factor.
