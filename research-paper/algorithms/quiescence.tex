\begin{algorithm2e}[tp]
\caption{Quiescence Search}
\label{alg:quiescence}
\DontPrintSemicolon
\Input{Position $p$, $\alpha$, $\beta$}
\Output{Quiescent score evaluation}
\BlankLine
$standPat \leftarrow \text{Evaluate}(p)$\;
\If{$standPat \ge \beta$}{
    \Return{$\beta$}\;
}
$\alpha \leftarrow \max(\alpha, standPat)$\;
\BlankLine
$captures \leftarrow \text{GenerateSortedCaptures}(p)$\;
\ForEach{capture move $c$ in $captures$}{
    $\text{MakeMove}(p, c)$\;
    $score \leftarrow - \text{QuiescenceSearch}(p, -\beta, -\alpha)$\;
    $\text{UnmakeMove}(p, c)$\;
    $\alpha \leftarrow \max(\alpha, score)$\;
    \If{$\alpha \ge \beta$}{
        \Return{$\beta$}\;
    }
}
\Return{$\alpha$}\;
\end{algorithm2e}
